      saveChecklistItems();
    }

    function addItemToCategory() {
      const name = itemNameInput.value.trim();
      const category = categorySelect.value;
      if (!name) {
        itemNameInput.focus();
        return;
      }
      if (!category) {
        categorySelect.focus();
        return;
      }
      if (!members.length) return;
      const { owners, isAllMembers } = readAddItemOwnerSelection();
      if (!owners.length) return;

      const card = categoryCards[category];
      if (!card) return;

      checklistItems.push({
        id: createId(),
        name,
        category,
        owners,
        isAllMembers,
        completedOwners: [],
        completed: false
      });
      saveChecklistItems();
      renderChecklistItems();
      syncGlobalsToCurrentTravel();
      saveTravels();
      renderHomePage();
      leaveAddItemPage();
    }
